======================ISOTROPIC CODE======================
0. LOAD MODULES:
module load intel/2020b

1.a. COMPILE STRING FOR ks_spectrum PARALLEL
COMPILER=intel MPP=true PRECISION=2 CTIME="-DNERSC_TIME -DCGTIME -DFFTIME -DFLTIME -DGFTIME -DREMAP -DPRTIME -DIOTIME" CDEBUG="-DREMAP_STDIO_APPEND -DHISQ_SVD_VALUES_INFO -DHISQ_SVD_COUNTER" make ks_spectrum_hisq

1.b. COMPILE STRING FOR ks_spectrum SCALAR
COMPILER=intel MY_CC=icc MY_CXX=icpc MPP=false PRECISION=2 CTIME="-DNERSC_TIME -DCGTIME -DFFTIME -DFLTIME -DGFTIME -DREMAP -DPRTIME -DIOTIME" CDEBUG="-DREMAP_STDIO_APPEND -DHISQ_SVD_VALUES_INFO -DHISQ_SVD_COUNTER" make ks_spectrum_hisq

Note: May need to be careful with the C compiler for "libraries",
e.g. set   CC = icc   in ../libraries/Make_vanilla
Note: by default MILC may use more modern OneAPI version of
intel compiler, icx, replace with icc.
